\begin{itemize}

	\item[---] ${R_i_j}$ –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ä–∞–¥–∏—É—Å–æ–≤ —Å–≤—è–∑–∏ –±–∞–∑–æ–≤–æ–π —Å—Ç–∞–Ω—Ü–∏–∏. –ü–∞—Ä–∞–º–µ—Ç—Ä $R_i_j$ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É–µ—Ç –¥–∞–ª—å–Ω–æ—Å—Ç—å —Å–≤—è–∑–∏ –º–µ–∂–¥—É —Å—Ç–∞–Ω—Ü–∏—è–º–∏ $a_i \in A_2$ –∏ $a_j \in A_2$, $i = \overline{n_1 + 1,n}, j = \overline{n_1 + 1, n}, i \ne j$;
	\item[---] $ùëñ\vartheta_i$ –æ–±—ä–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –≤ –µ–¥–∏–Ω–∏—Ü—É –≤—Ä–µ–º–µ–Ω–∏, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –±—ã—Ç—å –ø–æ–ª—É—á–µ–Ω –æ—Ç –æ–±–ª–∞—Å—Ç–µ–π, –æ–±—Å–ª—É–∂–∏–≤–∞–µ–º—ã—Ö –±–∞–∑–æ–≤–æ–π —Å—Ç–∞–Ω—Ü–∏–µ–π, –∏–ª–∏ –¥—Ä—É–≥–∏—Ö –±–∞–∑–æ–≤—ã—Ö —Å—Ç–∞–Ω—Ü–∏–π.
